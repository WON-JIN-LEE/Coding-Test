```javascript
function solution(num) {
  let cnt = 0;
  while (num != 1) {
    if (num % 2 == 0) num /= 2;
    else num = num * 3 + 1;
    cnt++;

    if (cnt > 500) return -1;
  }
  return cnt;
}
```

```javascript
let answer = 0;
let count = 0;

function solution(num) {
  if (answer === 1 || num === 1) {
    return count;
  }
  if (count > 500) {
    return -1;
  }

  if (num % 2 === 0) {
    answer = num / 2;
    count++;
    return solution(answer);
  } else {
    answer = num * 3 + 1;
    count++;
    return solution(answer);
  }
}
```
